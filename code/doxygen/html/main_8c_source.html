405 void Generate_Lookup_Tables(void) {
406  int16_t i;
407  uint32_t uacal, ua, urcal, ur;
408  int32_t lambda, temp;
409  float ip, o2, ri;
410  // Determine resistance calibration value for calculations further on
411  ur = optimal_resistance * VDDA * (365 + 187) / 365 / 4096;
412  urcal = -((15.5 * 0.000158 * 301) * 1000 - ur) * 17;
413 
414  // Determine lambda calibration value for calculations further on
415  uacal = optimal_lambda * VDDA * (365 + 187) / 365 / 4096;
416 
417  for (i = 0; i < 4096; i++) {
418  // Calculate the Lambda value
419  ua = i * VDDA * (365 + 187) / 365 / 4096; // mV
420  ip = (ua - uacal) * 1000 / (61.9 * 8); // mA
421  o2 = ip * 0.2095 / 2540;
422  lambda = (o2 / 3 + 1) / (1 - 4.77 * o2) * 1000;
423  if (lambda >= 10119 || lambda <= 0) {
424  lambda = 10119;
425  }
426  // Set lambda lookup table value
427  lambda_Lookup[i] = lambda;
428 
429  // Calculate the temperature value
430  ur = i * VDDA * (365 + 187) / 365 / 4096;
431  ri = (ur - urcal / 17) / (15.5 * 0.158);
432  temp = 4445 * pow(ri, -0.4449) + 428.6;
433  if (temp >= 7049|| temp <= 0) {
434  temp = 7049;
435  }
436  // Set temperature lookup table value
437  temp_Lookup[i] = temp;
438  }
