485 void Generate_Lookup_Tables(void) {
486  uint16_t i;
487  int32_t uacal, ua, urcal, ur, lambda, temp;
488  float ip, o2, ri;
489 
490  // Determine resistance calibration value for calculations further on
491  ur = optimal_resistance * VDDA * (365 + 187) / 365 / 4096;
492  urcal = -((15.5 * 0.000158 * 301) * 1000 - ur) * 17;
493 
494  // Determine lambda calibration value for calculations further on
495  uacal = optimal_lambda * VDDA * (365 + 187) / 365 / 4096;
496 
497  for (i = 0; i < 4096; i++) {
498  // Calculate the Lambda value
499  ua = i * VDDA * (365 + 187) / 365 / 4096;
500  ip = (ua - uacal) * 1000 / (61.9 * 8); // / 1000;
501  o2 = ip * 0.2095 / 2540;
502  lambda = (o2 / 3 + 1) / (1 - 4.77 * o2) * 1000;
503  //lambda = (492.3 * exp(-pow((ip - 3.869) / 0.772, 2)) + 2.183 * exp(-pow((ip - 2.288) / 0.714, 2)) + 1.09 * exp(-pow((ip - 2.8) / 6.656, 2)) + 1.011 * exp(-pow((ip - 1.697) / 1.112, 2))) * 1000;
504 
505  if (lambda >= 10119 || lambda <= 0) {
506  lambda = 10119;
507  }
508  // Set lambda lookup table value
509  lambda_Lookup[i] = lambda;
510 
511  // Calculate the temperature value
512  ur = i * VDDA * (365 + 187) / 365 / 4096;
513  ri = (ur - urcal / 17) / (15.5 * 0.158);
514  temp = 4445 * pow(ri, -0.4449) + 428.6;
515 
516  if (temp >= 7049|| temp <= 0) {
517  temp = 7049;
518  }
